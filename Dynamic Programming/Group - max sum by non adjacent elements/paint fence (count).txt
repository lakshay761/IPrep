 public int numWays(int n, int k) {
        // write your code here
        int[] same = new int[n];
        int[] diff = new int[n];
        same[0] = 0;
        diff[0] = k;
        for(int i = 1; i<n ; i++){
            diff[i] = ((k-1) * same[i-1]) + ((k-1) * diff[i-1]);
            same[i] = diff[i-1];
        }
        return diff[n-1] + same[n-1];
    }
