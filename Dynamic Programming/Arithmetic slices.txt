 public int numberOfArithmeticSlices(int[] A) {
  if(A.length <= 2){return 0;}
        int gap = A[1] - A[0];
        int[] dp = new int[A.length];
        int sum = 0;
        for(int i =2 ;i< A.length ; i++){
            if(A[i] - A[i-1] == gap){
                dp[i] = 1 + dp[i-1];
                sum += dp[i];            
            }else{
                gap = A[i] - A[i-1];
            }
        }
        return sum;
    }
