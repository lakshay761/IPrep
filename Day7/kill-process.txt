     public static void display(node pn, ArrayList<Integer> l) {
        for (node n: pn.children) {
            l.add(n.data);
            display(n, l);
        }
    }
     static class node{
         int data;
         ArrayList<node> children;
         node(int data){
             children= new ArrayList<>();
             this.data = data;
         }
     }
